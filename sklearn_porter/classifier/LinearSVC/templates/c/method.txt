int {name} (float atts[{n_features}]) {{

    {coefficients}
    {intercepts}

    double class_val = -INFINITY;
    int class_idx = -1;
    int i = 0, j = 0;
    for (i = 0; i < {n_classes}; i++) {{
        double prob = 0.;
        for (j = 0; j < {n_features}; j++) {{
            prob += coefs[i][j] * atts[j];
        }}
        if (prob + inters[i] > class_val) {{
            class_val = prob + inters[i];
            class_idx = i;
        }}
    }}
    return class_idx;
}}
